#define THINGER_SERIAL_DEBUG
#include <ThingerESP32.h>
#include "DHTesp.h"
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <EEPROM.h>

// ==== Thinger.io ====
#define USERNAME "felipepizzinato"
#define DEVICE_ID "esp32_dht22"
#define DEVICE_CREDENTIAL "372mZ15!zFYM_l7F"

// ==== Wi-Fi ====
#define SSID "Wokwi-GUEST"
#define SSID_PASSWORD ""

// ==== Pinos (ajustados para ESP32) ====
#define DHT_PIN 15
#define LED_PIN 26
#define BTN_PIN 14
#define BUZZER_PIN 27

// ==== Objetos ====
DHTesp dht;
LiquidCrystal_I2C lcd(0x27, 16, 2);
ThingerESP32 thing(USERNAME, DEVICE_ID, DEVICE_CREDENTIAL);

// ==== Estado ====
bool deviceState = false;
bool lastButton = HIGH;
unsigned long lastUpdate = 0;
const int EEPROM_ADDR = 0;

// ==== Variáveis de leitura ====
float temp = 0.0;
float hum  = 0.0;

void setup() {
  Serial.begin(115200);
  delay(200);

  // === Wi-Fi e Thinger ===
  thing.add_wifi(SSID, SSID_PASSWORD);
  Serial.println("Conectando ao Thinger...");

  // === Sensores e atuadores ===
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);
  pinMode(BTN_PIN, INPUT_PULLUP); // botão com pullup interno

  dht.setup(DHT_PIN, DHTesp::DHT22);
  lcd.init();
  lcd.backlight();

  EEPROM.begin(4);
  deviceState = EEPROM.read(EEPROM_ADDR) == 1;

  // === Recursos Thinger.io ===
  thing["GPIO_26"] << digitalPin(LED_PIN);

thing["dht"] >> [](pson &out){
  out["temp"] = temp;
  out["hum"]  = hum;
};



// WRITE: recebe do dashboard e aplica
thing["state"] << [](pson &in){
  if(!in.is_empty()){
    deviceState = (bool) in;
    EEPROM.write(EEPROM_ADDR, deviceState ? 1 : 0);
    EEPROM.commit();

    digitalWrite(LED_PIN, deviceState);
    digitalWrite(BUZZER_PIN, deviceState);

    Serial.print("Estado alterado via dashboard -> ");
    Serial.println(deviceState ? "ON" : "OFF");
  }
};

// READ: expõe o valor para widgets (Led Indicator, etc.)
thing["state"] >> [](pson &out){
  out = deviceState;
};


  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("System ready!");
  delay(1000);
  lcd.clear();
}

void loop() {
  // === Botão (com debounce simples) ===
  int buttonValue = digitalRead(BTN_PIN);
  if (buttonValue == LOW && lastButton == HIGH) {
    deviceState = !deviceState;
    EEPROM.write(EEPROM_ADDR, deviceState ? 1 : 0);
    EEPROM.commit();

    Serial.print("Button pressed -> Status: ");
    Serial.println(deviceState ? "ON" : "OFF");

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Status:");
    lcd.setCursor(8, 0);
    lcd.print(deviceState ? "ON " : "OFF");

    delay(200);
  }
  lastButton = buttonValue;

  // === Saídas físicas ===
  digitalWrite(LED_PIN, deviceState);
  digitalWrite(BUZZER_PIN, deviceState);

  // === Leitura do DHT a cada 10s ===
  if (millis() - lastUpdate >= 10000) {
    lastUpdate = millis();

    TempAndHumidity data = dht.getTempAndHumidity();
    temp = data.temperature;
    hum  = data.humidity;

    if (isnan(temp) || isnan(hum)) {
      Serial.println("Sensor error!");
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Read fail!");
    } else {
      Serial.printf("Temp: %.1f °C | Umid: %.1f %%\n", temp, hum);

      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("T:");
      lcd.print(temp, 1);
      lcd.print("C");
      lcd.setCursor(0, 1); 
      lcd.print("H:");
      lcd.print(hum, 1);
      lcd.print("%");
    }
  }

  // === Comunicação com o Thinger ===
  thing.handle();
  delay(50);
}
