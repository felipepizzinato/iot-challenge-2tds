#include "DHT.h"
#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <EEPROM.h>   // biblioteca para salvar estado

#define DHTPIN 2        
#define DHTTYPE DHT22   
#define LED_PIN 10
#define BTN_PIN 11
#define BUZZER_PIN 9

bool deviceState = false;         
bool lastButton = HIGH;          
unsigned long lastUpdate = 0;     

DHT dht(DHTPIN, DHTTYPE);
LiquidCrystal_I2C lcd(0x27, 16, 2);

// endereço na EEPROM para salvar estado
const int EEPROM_ADDR = 0;

void setup() {
  Serial.begin(9600);

  pinMode(BTN_PIN, INPUT);   // botão com resistor físico
  pinMode(LED_PIN, OUTPUT);
  pinMode(BUZZER_PIN, OUTPUT);

  dht.begin();

  lcd.init();
  lcd.backlight();

  // Recupera estado salvo na EEPROM
  byte savedState = EEPROM.read(EEPROM_ADDR);
  if (savedState == 1) {
    deviceState = true;
  } else {
    deviceState = false;
  }

  // Mostra status inicial no LCD
  lcd.setCursor(0, 0);
  lcd.print("System ready!");
  delay(1000);
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print("Status:");
  lcd.setCursor(8, 0);
  lcd.print(deviceState ? "ON " : "OFF");

  Serial.println("System ready!");
}

void loop() {
  int buttonValue = digitalRead(BTN_PIN);

  if (buttonValue == LOW && lastButton == HIGH) {
    // alterna estado
    deviceState = !deviceState;

    // salva na EEPROM
    EEPROM.write(EEPROM_ADDR, deviceState ? 1 : 0);

    Serial.print("Button pressed -> Status: ");
    Serial.println(deviceState ? "ON" : "OFF");

    lcd.clear();
    lcd.setCursor(0, 0);
    lcd.print("Status:");
    lcd.setCursor(8, 0);
    lcd.print(deviceState ? "ON " : "OFF");

    delay(200); // debounce
  }

  lastButton = buttonValue;

  // Saídas
  digitalWrite(LED_PIN, deviceState ? HIGH : LOW);
  digitalWrite(BUZZER_PIN, deviceState ? HIGH : LOW);

  // leitura do DHT a cada 10s
  if (millis() - lastUpdate >= 10000) {
    lastUpdate = millis();

    float tempC = dht.readTemperature();
    float hum = dht.readHumidity();

    if (isnan(tempC) || isnan(hum)) {
      Serial.println("Sensor error!");
      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("Read fail!");
    } else {
      Serial.print("Temp: "); Serial.print(tempC); Serial.println(" C");
      Serial.print("Humidity: "); Serial.print(hum); Serial.println(" %");

      lcd.clear();
      lcd.setCursor(0, 0);
      lcd.print("T:"); lcd.print(tempC); lcd.print("C");

      lcd.setCursor(0, 1);
      lcd.print("H:"); lcd.print(hum); lcd.print("%");
    }
  }

  delay(40);
}
